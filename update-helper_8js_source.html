327  * @param {string} v1
328  * @param {string} v2
329  * @return {number} 1 if v1 > v2, -1 if v1 < v2, 0 if v1 === v2
330  * @details This function will compare two version numbers and return 1 if v1 > v2, -1 if v1 < v2, 0 if v1 === v2
331  * compareVersions('1.0.0', '1.0.1'); // -1
332  * compareVersions('1.0.1', '1.0.0'); // 1
333  * compareVersions('1.0.0', '1.0.0'); // 0
334  */
335  function compareVersions(v1, v2) {
336  const v1Parts = v1.split('.').map(Number);
337  const v2Parts = v2.split('.').map(Number);
338  const maxLength = Math.max(v1Parts.length, v2Parts.length);
339 
340  for (let i = 0; i < maxLength; i++) {
341  const v1Part = v1Parts[i] || 0;
342  const v2Part = v2Parts[i] || 0;
343 
344  if (v1Part > v2Part) {
345  return 1;
346  }
347  if (v1Part < v2Part) {
348  return -1;
349  }
350  }
351  return 0;
352  }
