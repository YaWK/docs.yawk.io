159  * @param object $db Database object
160  * @return bool

162  static function checkLogin($db)
163  { /** @var $db \YAWK\db */
164  /* check user login */
165  $user = new \YAWK\user($db);
166  if(isset($_POST['user']) && isset($_POST['password'])) {

168  { // create session var
169  $_SESSION['username'] = $_POST['user'];
170  $_SESSION['passwordFail'] = 0;
171  $user->storeLogin($db, 0, "backend", $_POST['user'], $_POST['password']);
172  return true;
173  }
174  else
175  { // if username or pwd is wrong
176  if (isset($_SESSION['passwordFail']))
177  { // add password fail counter
178  $_SESSION['passwordFail']++;
179  }
180  else
181  { // first wrong try
182  $_SESSION['passwordFail'] = 1;
183  }
184  // log this user login / store @ login db table
185  $user->storeLogin($db, 1, "backend", $_POST['user'], $_POST['password']);
186  return false;

192  }
