365  /**
366  * Returns all available folders
367  *
368  * @param string $separator default is '.'
369  * @param integer $type has three meanings 0,1,2.
370  * If 0 return nested array, if 1 return an array of strings, if 2 return raw imap_list()
371  * @return array with folder names
372  */
373  public function getFolders($separator = null, $type = 0)
374  {
375  if(preg_match( '/^{.+}/', $this->mailbox, $matches)){
376  $mailbox = $matches[0];
377  }else{
378  $mailbox = $this->mailbox;
379  };
380 
381  $folders = imap_list($this->imap, $mailbox, "*");
382 
383  if ($type == 2) {
384  return $folders;
385  };
386  if ($type == 1) {
387  return str_replace($mailbox, "", $folders);
388  };
389  if ($type == 0) {
390  $arrayRaw = str_replace($mailbox, "", $folders);
391  if (!isset($separator)) {
392  $separator = '.';
393  };
394  $arrayNew = [];
395  foreach ($arrayRaw as $string) {
396  $array = explode($separator, $string);
397  $count = count($array);
398  $count = $count-1;
399  $cache = false;
400  for($i=$count; $i>=0; $i--){
401  if($i == $count){
402  $cache = [$array[$i]=>[]];
403  }else{
404  $cache = [$array[$i] => $cache];
405  };
406  };
407  $arrayNew = array_merge_recursive($arrayNew, $cache);
408  };
409  return $arrayNew;
410  }
411  return null;
412  }
