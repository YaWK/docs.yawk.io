127  if (code <= 0xf7) return lowTypes.charAt(code)
128  else if (0x590 <= code && code <= 0x5f4) return "R"
129  else if (0x600 <= code && code <= 0x6ed) return arabicTypes.charAt(code - 0x600)
130  else if (0x6ee <= code && code <= 0x8ac) return "r"
131  else if (0x2000 <= code && code <= 0x200b) return "w"
132  else if (code == 0x200c) return "b"
133  else return "L"
134  }

142  this.level = level
143  this.from = from; this.to = to

146  return function(str) {
147  if (!bidiRE.test(str)) return false
148  let len = str.length, types = []
149  for (let i = 0; i < len; ++i)
150  types.push(charType(str.charCodeAt(i)))

152  // W1. Examine each non-spacing mark (NSM) in the level run, and
153  // change the type of the NSM to the type of the previous
154  // character. If the NSM is at the start of the level run, it will
155  // get the type of sor.
156  for (let i = 0, prev = outerType; i < len; ++i) {
157  let type = types[i]
158  if (type == "m") types[i] = prev
159  else prev = type
160  }
161 
162  // W2. Search backwards from each instance of a European number
163  // until the first strong type (R, L, AL, or sor) is found. If an
164  // AL is found, change the type of the European number to Arabic
165  // number.
166  // W3. Change all ALs to R.
167  for (let i = 0, cur = outerType; i < len; ++i) {
168  let type = types[i]
169  if (type == "1" && cur == "r") types[i] = "n"
170  else if (isStrong.test(type)) { cur = type; if (type == "r") types[i] = "R" }
171  }
172 
173  // W4. A single European separator between two European numbers
174  // changes to a European number. A single common separator between
175  // two numbers of the same type changes to that type.
176  for (let i = 1, prev = types[0]; i < len - 1; ++i) {
177  let type = types[i]
178  if (type == "+" && prev == "1" && types[i+1] == "1") types[i] = "1"
179  else if (type == "," && prev == types[i+1] &&
180  (prev == "1" || prev == "n")) types[i] = prev
181  prev = type
182  }
183 
184  // W5. A sequence of European terminators adjacent to European
185  // numbers changes to all European numbers.
186  // W6. Otherwise, separators and terminators change to Other
187  // Neutral.
188  for (let i = 0; i < len; ++i) {
189  let type = types[i]
190  if (type == ",") types[i] = "N"
191  else if (type == "%") {
192  let end
193  for (end = i + 1; end < len && types[end] == "%"; ++end) {}
194  let replace = (i && types[i-1] == "!") || (end < len && types[end] == "1") ? "1" : "N"
195  for (let j = i; j < end; ++j) types[j] = replace
196  i = end - 1
197  }
198  }

200  // W7. Search backwards from each instance of a European number
201  // until the first strong type (R, L, or sor) is found. If an L is
202  // found, then change the type of the European number to L.
203  for (let i = 0, cur = outerType; i < len; ++i) {
204  let type = types[i]
205  if (cur == "L" && type == "1") types[i] = "L"
206  else if (isStrong.test(type)) cur = type
207  }
208 
209  // N1. A sequence of neutrals takes the direction of the
210  // surrounding strong text if the text on both sides has the same
211  // direction. European and Arabic numbers act as if they were R in
212  // terms of their influence on neutrals. Start-of-level-run (sor)
213  // and end-of-level-run (eor) are used at level run boundaries.
214  // N2. Any remaining neutrals take the embedding direction.
215  for (let i = 0; i < len; ++i) {
216  if (isNeutral.test(types[i])) {
217  let end
218  for (end = i + 1; end < len && isNeutral.test(types[end]); ++end) {}
219  let before = (i ? types[i-1] : outerType) == "L"
220  let after = (end < len ? types[end] : outerType) == "L"
221  let replace = before || after ? "L" : "R"
222  for (let j = i; j < end; ++j) types[j] = replace
223  i = end - 1
224  }

228  // building up an actual levels array. Since there are only three
229  // levels (0, 1, 2) in an implementation that doesn't take
230  // explicit embedding into account, we can build up the order on
231  // the fly, without following the level-based algorithm.
232  let order = [], m

246  order.splice(at, 0, new BidiSpan(2, nstart, j))
247  pos = j
248  } else ++j
249  }
250  if (pos < i) order.splice(at, 0, new BidiSpan(1, pos, i))
251  }
252  }

254  order[0].from = m[0].length
255  order.unshift(new BidiSpan(0, 0, m[0].length))
256  }

264  order.push(new BidiSpan(order[0].level, len, len))
